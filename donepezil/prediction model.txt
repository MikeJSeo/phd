model{
  
  for(i in 1:Np){
    y[i] ~ dnorm(mu[i], tau)
    mu[i] <- a[studyid[i]] + inprod(beta[], X[i,])
  }
  tau ~ dgamma(0.001, 0.001)
  sigma <- pow(tau, -1/2)
 
  alpha ~ dnorm(0,0.001)
  for(j in 1:Nstudies){
    a[j] ~ dnorm(alpha, tau.alpha)
  }
  tau.alpha <- pow(sd.alpha, -2)
  sd.alpha ~ dunif(0, 100)

  for(k in 1:Ncovariate){
    beta[k] ~ dnorm(0, 0.001)
  }
  
}
