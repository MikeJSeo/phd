model{

#likelihood
y1 ~ dmnorm(Mu1, Omega1[,])
y2 ~ dmnorm(Mu2, Omega2[,])
y3 ~ dmnorm(Mu3, Omega3[,])

Mu1 <- c(alpha, beta[], gamma2[], gamma3[])
Mu2 <- c(alpha, beta[], gamma2[])
Mu3 <- c(alpha, beta[], gamma3[])


#parameters
alpha ~ dnorm(0, 0.001)
for(i in 1:ncovariate){
	beta[i] ~ dnorm(0, 0.001)
}

for(i in 1:ncovariate){
	gamma2[i] ~ dnorm(0, 0.001)
}

for(i in 1:ncovariate){
	gamma3[i] ~ dnorm(0, 0.001)
}



}
