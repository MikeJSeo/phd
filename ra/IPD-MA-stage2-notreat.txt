model{

#likelihood
y1 ~ dmnorm(Mu, Omega1[,])
y2 ~ dmnorm(Mu, Omega2[,])
y3 ~ dmnorm(Mu3, Omega3[,])
y4 ~ dmnorm(Mu4, Omega4[,])

Mu <- c(alpha, beta[], gamma2[], gamma3[])
Mu3 <- c(alpha, beta[], gamma2[])
Mu4 <- c(alpha, beta[], gamma3[])


#parameters
alpha ~ dnorm(0, 0.001)
for(i in 1:ncovariate){
	beta[i] ~ dnorm(0, 0.001)
}

for(i in 1:ncovariate){
	gamma2[i] ~ dnorm(0, 0.001)
}

for(i in 1:ncovariate){
	gamma3[i] ~ dnorm(0, 0.001)
}



}
